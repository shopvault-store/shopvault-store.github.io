Zadatak 4 [srednji]

Napi≈°i program koji dati hex broj u char vrednosti pomoƒáu funkcije hex_to_int konvertuje u decimalni broj pa se onda ispisuje na konzolu kao u pro≈°lom zadatku (dakle int broj i printf("%c",broj))


#include <stdio.h>
#include <string.h>

int hex_to_int(const char *hex) {
    int rezultat = 0;
    int i = 0;

    // Ako poƒçinje sa "0x" ili "0X", preskoƒçi
    if (strlen(hex) > 2 && hex[0] == '0' && (hex[1] == 'x' || hex[1] == 'X')) {
        i = 2;
    }

    for (; i < strlen(hex); i++) {
        char c = hex[i];
        int vrednost;

        if (c >= '0' && c <= '9') {
            vrednost = c - '0';
        } else if (c >= 'A' && c <= 'F') {
            vrednost = c - 'A' + 10;
        } else if (c >= 'a' && c <= 'f') {
            vrednost = c - 'a' + 10;
        } else {
            break; // neva≈æeƒái znak
        }

        rezultat = rezultat * 16 + vrednost;
    }

    return rezultat;
}

int main(void) {
    const char *rec = "0x65";
    int broj = hex_to_int(rec);

    printf("Decimalno: %d\n", broj);   // 101
    printf("Kao karakter: %c\n", broj); // e

    return 0;
}

Napi≈°i program koji ispisuje ono ≈°to je napisano u heksadecimalnom zapisu (dakle npr za slova iz ASCII tabele koristimo format NN a ne format 0xNN sto se tice input-a u program). Podsetimo se da je 'I' = 0x49 i da je 's' = 0x73 u ASCII tabeli.

Za ulaz: 497370697420555550
treba da dobijemo izlaz: Ispit UUP 


#include <stdio.h>
#include <stdlib.h>

// 0x49 = 'I'
// getchar vraca int a ne char
//(dakle za vrednost '4' vraca decimalno 52)
//(za '4' vraca hex vrednost = 0x34)

// ulaz: 497370697420555550
// izlaz: Ispit UUP

int hexval(int c){
    if(c >= '0' && c <= '9') return c - '0';
    if(c >= 'a' && c <= 'f') return 10 + (c - 'a');
    if(c >= 'A' && c <= 'F') return 10 + (c - 'A');
    return -1; //nevazeci znak
}

int main(){
    int niz[128];
    int n = 0;
    int c1, c2;

    while(1){
        c1 = getchar();
        if(c1 == '\n' || c1 == EOF ){
            break;
        }
        c2 = getchar();
        if(c2 == '\n' || c2 == EOF){
            break;
        }

        int hi = hexval(c1);
        int lo = hexval(c2);
        if(hi < 0 || lo < 0){
            break; //ako se naidje na neki nevazeci znak
        }

        niz[n++] = (hi << 4) | lo;  //dodajemo "lo" na "hi"

        /*
        hi      = 0x04   // 0000 0100

        hi << 4 = 0x40   // 0100 0000
        lo      = 0x09   // 0000 1001
        -------------------------------- OR
        rezultat = 0x49  // 0100 1001

        */

    }

    for(int i = 0; i < n; i++){
        //putchar((char)niz[i]); //moze i ovako
        printf("%c", niz[i]);
    }
    putchar('\n');




return 0;
}
